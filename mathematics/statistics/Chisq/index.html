Chisq distributions

χ2\chi^2χ2 distribution

  • The Chi squared distribution χ2(k)\chi^2(k)χ2(k) with some integer kkk degrees of freedom is the sum of squares of kkk independent standard Normal variates.

using Distributions 
using Plots
using SpecialFunctions
using LaTeXStrings
@doc Chisq

Prepare standard normal distributions

d = Normal(0,1)

Plot histogram

sample = rand(d, 10000)
p = plot(xlim=[-3, 3])
histogram!(sample, normalize=:pdf, label="hist")
plot!(x->pdf(d, x),label="pdf")
savefig(joinpath(@OUTPUT, "normal.svg"))

Prepare Γ\GammaΓ function

@doc gamma
Γ(x) = gamma(x);
function T(n, z) 
    if z > 0
        a = z^(n/2 - 1)*exp(-z/2)
        b = 2^(n/2) * Γ(n/2)
        return a/b
    else
        return 0.
    end
end;

Generate χ2\chi^2χ2 distribution with ...

degree = 1

n_trial = 100000
X = rand(d, n_trial)
X² = X .^ 2.
p = plot()
histogram!(p, X², normalize=:pdf, label=L"X^2")
x = range(0.01,2,length=100)
y = T.(1, x)
plot!(p, x, y, xlim=[0,2], ylim=[0,4], label=L"T_1")

y = pdf.(Chisq(1), x)
plot!(p, x, y, line=:dash, label=L"\chi^2(1)") ;

y = pdf.(χ², x)
χ² = Chisq(2)
plot!(p, x, y, label=L"\chi^2(2)",line=:dash)
