Rust

Hello Rust

  • You can compile Rust code and run executable file from Julia with the following code:

run(`rustc --version`)
rustc 1.58.1 (db9d1b20b 2022-01-20)

rscode = raw"""
fn main(){
	println!("Hello Rust from Julia runtime");
}
"""

open(`rustc -o sample -`, "w") do f
    print(f, rscode)
end

run(`./sample`)

fn main(){
    println!("Hello Rust from Julia runtime");
}
