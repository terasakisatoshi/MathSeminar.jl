GR.jl

This example is taken from https://mybinder.org/v2/gh/jheinen/GR.jl/master

using GR
using LaTeXStrings
GR.inline("mov")
x = collect(0:0.01:2*pi)
for i = 1:100
    GR.plot(x, sin.(x .+ i / 10.0), size=(400,300))
end

s = GR.show()
